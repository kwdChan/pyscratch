 190    def transform_frames(self):
 191        self.request_transform = False
 192        for k, frames in self.frame_dict_original.items():
 193            new_frames = []
 194            for idx, f in enumerate(frames):
 195                f_new = f.copy()
 196                for s, o in self.blit_surfaces[(k, idx)]:
 197                    f_new.blit(s, o)
 198                f_new = set_transparency(f_new, self.transparency_factor)
 199                f_new = adjust_brightness(f_new, self.brightness_factor)
 200                f_new = pygame.transform.scale_by(f_new, self.scale_factor)
 201                new_frames.append(f_new)
 202
 203            self.frame_dict[k] = new_frames

 205        self.frames = self.frame_dict[self.frame_mode]

 207    def on_update(self, x, y, angle) -> Tuple[pygame.Surface, pygame.Rect]:
 208        if self.request_transform:
 209            self.transform_frames()
 210
 211        img = self.frames[self.frame_idx]
 212
 213        if self.rotation_style == _RotationStyle.ALL_AROUND: 
 214            img = pygame.transform.rotate(img, -angle)
 215    
 216        elif self.rotation_style == _RotationStyle.LEFTRIGHT:
 217            if angle > -90 and angle < 90:
 218                pass
 219            else:
 220                img = pygame.transform.flip(img, True, False)
 221             
 222        elif self.rotation_style == _RotationStyle.FIXED:
 223            pass
 224
 225        img = pygame.transform.flip(img, self.flip_x, self.flip_y)
 226
 227        self.image = img
 228
 229        img_w, img_h = img.get_width(), img.get_height()
 230        rect = img.get_rect(
 231            center=(x, y),
 232              width=img_w,
 233              height=img_h,
 234              )
 235        return img, rect

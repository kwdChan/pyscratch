 16def collision_begin(arbiter, space, data):
 17    game = cast(Game, data['game'])
 18    game.contact_pairs_set.add(arbiter.shapes) 
 19
 20    for e, (a,b) in game.trigger_to_collision_pairs.items():
 21        if (a.shape in arbiter.shapes) and (b.shape in arbiter.shapes):
 22            e.trigger(arbiter)
 23
 24
 25    colliding_types = arbiter.shapes[0].collision_type, arbiter.shapes[1].collision_type
 26    collision_allowed = True
 27    for collision_type, (allowed, triggers) in game.collision_type_to_trigger.items():
 28        if collision_type in colliding_types: 
 29            [t.trigger(arbiter) for t in triggers]
 30            collision_allowed = collision_allowed and allowed
 31        
 32
 33    if (arbiter.shapes[0].collision_type == 0) or (arbiter.shapes[1].collision_type == 0):
 34        collision_allowed = False
 35
 36
 37    return collision_allowed

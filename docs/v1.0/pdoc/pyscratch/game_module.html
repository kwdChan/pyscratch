  23def _collision_begin(arbiter, space, data):
  24    game = cast(Game, data['game'])
  25    game._contact_pairs_set.add(arbiter.shapes) 
  26
  27    for e, (a,b) in game._trigger_to_collision_pairs.items():
  28        if (a._shape in arbiter.shapes) and (b._shape in arbiter.shapes):
  29            e.trigger(arbiter)
  30
  31
  32    colliding_types = arbiter.shapes[0].collision_type, arbiter.shapes[1].collision_type
  33    collision_allowed = True
  34    for collision_type, (allowed, triggers) in game._collision_type_to_trigger.items():
  35        if collision_type in colliding_types: 
  36            [t.trigger(arbiter) for t in triggers]
  37            collision_allowed = collision_allowed and allowed
  38        
  39
  40    if (arbiter.shapes[0].collision_type == 0) or (arbiter.shapes[1].collision_type == 0):
  41        collision_allowed = False
  42
  43
  44    return collision_allowed

  46def _collision_separate(arbiter, space, data):
  47    game = cast(Game, data['game'])
  48
  49    if arbiter.shapes in game._contact_pairs_set:
  50        game._contact_pairs_set.remove(arbiter.shapes)

  53    reverse_order = arbiter.shapes[1], arbiter.shapes[0]
  54    if reverse_order in game._contact_pairs_set:
  55        game._contact_pairs_set.remove(reverse_order)

 304    def start(self, framerate, sim_step_min=300, debug_draw=False, event_count=False):
 305        """
 306        Start the game. 
 307
 308        Parameters
 309        ---
 310        framerate : int
 311            The number of frames per second
 312
 313        sim_step_min: int
 314            The number of physics steps per second. Increase this value if the physics is unstable and decrease it if the game runs slow.
 315        
 316        debug_draw: bool
 317            Whether or not to draw the collision shape for debugging purposes
 318
 319        event_count: bool
 320            Whether or not to print out the number of active events for debugging purposes
 321
 322        """
 323
 324
 325        clock = pygame.time.Clock()
 326
 327        draw_every_n_step = sim_step_min//framerate+1
 328
 329        self._current_time_ms = 0
 330
 331
 332        for t in self._game_start_triggers:
 333            t.trigger()
 334
 335        while True:
 336            dt = clock.tick(framerate)
 337            self._current_time_ms += dt
 338            for i in range(draw_every_n_step): 
 339                self._space.step(dt/draw_every_n_step)
 340
 341
 342
 343            self._all_pygame_events = pygame.event.get()
 344            for event in self._all_pygame_events:
 345                if event.type == pygame.QUIT:
 346                    return 
 347
 348
 349            # for j in self.all_forever_jobs:
 350            #     j()
 351
 352            # check conditions
 353            for c in self._all_conditions:
 354                c._check()
 355
 356            # execute 
 357            for t in self._all_triggers:
 358                t._handle_all(self._current_time_ms)
 359                # TODO: is it possible to remove t in the self.all_triggers here?
 360                t._generators_proceed(self._current_time_ms)
 361
 362            # clean up
 363            self._all_conditions = list(filter(lambda t: t.stay_active, self._all_conditions))
 364            self._all_simple_key_triggers = list(filter(lambda t: t.stay_active, self._all_simple_key_triggers))
 365            self._all_triggers = list(filter(lambda t: t.stay_active, self._all_triggers))
 366
 367            
 368            if event_count: 
 369                print("all_conditions", len(self._all_conditions))
 370                print("all_triggers", len(self._all_triggers))
 371                print("all sprite", len(self._all_sprites))
 372                # print("all_simple_key_triggers", len(self.all_simple_key_triggers))
 373
 374            # Drawing
 375
 376            self._screen.fill((30, 30, 30))
 377            if not (self.__backdrop_index is None): 
 378                self._screen.blit(self.backdrops[self.__backdrop_index], (0, 0))
 379
 380            if debug_draw: 
 381                self._space.debug_draw(self._draw_options)
 382            
 383            self._all_sprites.update(self._space)
 384            self._all_sprites_to_show.draw(self._screen)
 385            pygame.display.flip()

 305    def start(self, framerate, sim_step_min=300, debug_draw=False, event_count=False):
 306        """
 307        Start the game. 
 308
 309        Parameters
 310        ---
 311        framerate : int
 312            The number of frames per second
 313
 314        sim_step_min: int
 315            The number of physics steps per second. Increase this value if the physics is unstable and decrease it if the game runs slow.
 316        
 317        debug_draw: bool
 318            Whether or not to draw the collision shape for debugging purposes
 319
 320        event_count: bool
 321            Whether or not to print out the number of active events for debugging purposes
 322
 323        """
 324
 325
 326        clock = pygame.time.Clock()
 327
 328        draw_every_n_step = sim_step_min//framerate+1
 329
 330        self._current_time_ms = 0
 331
 332
 333        for t in self._game_start_triggers:
 334            t.trigger()
 335
 336        while True:
 337            dt = clock.tick(framerate)

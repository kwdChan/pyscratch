  25def create_animated_sprite(folder_path,  *args, **kwargs):
  26    """
  27    Create a sprite using the images inside a folder. 
  28
  29    The folder should be organised in one of these two ways: 
  30
  31    **Option 1**: Only one frame mode. *Note that the images must be numbered.*
  32    ```
  33    ├─ player/
  34        ├─ 0.png
  35        ├─ 1.png
  36        ├─ 2.png
  37        ...
  38    ```
  39
  40    **Option 2**: Multiple frame modes. *Note that the images must be numbered.*
  41    ```
  42    ├─ player/  
  43        ├─ walking/  
  44            ├─ 0.png  
  45            ├─ 1.png  
  46            ...
  47        ├─ idling/
  48            ├─ 0.png
  49            ├─ 1.png
  50            ... 
  51        ...
  52    ``` 
  53
  54
  55    **Example**
  56    ```python
  57    # takes the path of the folder 
  58    my_sprite1 = create_animated_sprite("assets/player") 
  59    
  60    # optionally take in whatever parameters that the `Sprite` constructor take. 
  61    my_sprite2 = create_animated_sprite("assets/player", position=(200, 200)) 
  62    
  63    # For option 2 only: optionally set the `starting_animation` parameter in the `Sprite` constructor, but it'd still be fine without it.
  64    my_sprite3 = create_animated_sprite("assets/player", "idling")
  65    
  66
  67    # Event: to animate the sprite
  68    def animating():
  69    
  70        # switch to the next frame every 0.2 second
  71        while True: 
  72            my_sprite1.next_frame() # scratch block: next costume
  73            yield 0.2
  74
  75    my_sprite1.when_game_start().add_handler(animating)
  76
  77
  78    # if doing option 2: use `set_animation` to select the frame
  79    # there's no equivalent scratch block
  80    def movement():
  81        while True: 
  82            if pysc.sensing.is_key_pressed('right'): 
  83                my_sprite1.set_animation('walking') # reference to the folder name
  84                my_sprite1.x += 1
  85            else:
  86                my_sprite1.set_animatione('idling') 
  87            
  88            yield 1/FRAMERATE
  89
  90    my_sprite1.when_game_start().add_handler(movement)
  91
  92    ```
  93    Parameters
  94    ---
  95    folder_path: str
  96        The path of the folder that contains the images
  97
  98    \*args & \*\*kwargs: Optional
  99        Whatever the `Sprite` constructor takes, except the `frame_dict` parameter.
 100    """
 101    frame_dict = load_frames_from_folder(folder_path)
 102    return Sprite(frame_dict, *args, **kwargs)

 130def create_shared_data_display_sprite(key, font, size = (150, 50), bg_colour=(127, 127, 127), text_colour=(255,255,255), position: Optional[Tuple[float, float]]=None, update_period=0.1, **kwargs):
 131    """
 132    Create a display for a variable inside shared_data given the dictionary key (i.e. the name of the variable). 
 133    The variable display will update every `update_period` seconds.
 134    The variable display is created as a sprite. 
 135
 136    This function is created using only basic pyscratch functions and events. 
 137    If you want a more customised display, you may want to have a look at the source code as an example how it's done.
 138
 139    Example
 140    ```python
 141    
 142    # if the font is shared by multiple sprites, consider putting it in `settings.py`
 143    font = pygame.font.SysFont(None, 48)  # None = default font, 48 = font size
 144
 145    # the shared data
 146    game.shared_data['hp'] = 10
 147    
 148    # the variable display is created as a sprite 
 149    health_display = create_shared_data_display_sprite('hp', font, position=(100, 100), update_period=0.5)
 150    
 151    ```
 152    Parameters
 153    ---
 154    key: str
 155        The dictionary key of the variable in `game.shared_data` that you want to display.
 156    font: pygame.font.Font
 157        The pygame font object. Refer to the website of pygame for more details.
 158    size: Tuple[float, float]
 159        The size of the display panel
 160    bg_colour: Tuple[int, int, int] or Tuple[int, int, int, int] 
 161        The colour of the display panel in RGB or RGBA. Value range: [0-255]
 162    text_colour: Tuple[int, int, int] or Tuple[int, int, int, int] 
 163        The colour of the text in RGB or RGBA. Value range: [0-255]
 164    position: Tuple[int, int] 
 165        The position of the display
 166    update_period: float
 167        The variable display will update every `update_period` seconds.
 168    \*\*kwargs: Optional
 169        Whatever the `Sprite` constructor takes, except `frame_dict`,`starting_animation` & `position`
 170    """
 171
 172    w, h = size
 173    if position is None:
 174        position = w/2+25, h/2+25
 175    sprite = create_rect_sprite(bg_colour, w, h, position=position, **kwargs)
 176
 177    def update_value():
 178        while True: 
 179            sprite.write_text(f"{key}: {game.shared_data[key]}", font=font, colour=text_colour, offset=(w/2, h/2))
 180            yield update_period
 181
 182    sprite.when_game_start().add_handler(update_value)
 183    return sprite

 270class _DrawingManager:
 271    def __init__(self, frame_dict, starting_animation):
 272
 273        
 274        self.frame_dict_original: _FrameDictType = {k: [i.copy() for i in v] for k, v in frame_dict.items()} # never changed
 275        self.frame_dict: _FrameDictType = {k: [i.copy() for i in v] for k, v in frame_dict.items()} # transformed on the spot
 276        
 277
 278        self.animation_name = starting_animation
 279        self.frames = self.frame_dict[self.animation_name]
 280        self.frame_idx: int = 0
 281
 282        # transforming parameters -> performed during update, but only when the transform is requested
 283        self.request_transform = False
 284        self.transparency_factor = 1.0
 285        self.brightness_factor = 1.0
 286        self.scale_factor: float = 1.0
 287        self.rotation_offset: float # TODO: to be implemented
 288
 289        def create_blit_surfaces():
 290            blit_surfaces = {}
 291            for k in self.frame_dict_original:
 292                for i in range(len(self.frame_dict_original[k])):
 293                    blit_surfaces[(k, i)] = []
 294            return blit_surfaces
 295        self.blit_surfaces: Dict[Tuple[str, int], List[Tuple[pygame.Surface, Tuple[float, float]]]] = create_blit_surfaces()
 296
 297
 298        # rotation and flips -> performed every update on the current frame
 299        self.rotation_style: _RotationStyle = _RotationStyle.ALL_AROUND
 300        self.flip_x: bool = False
 301        self.flip_y: bool = False
 302
 303    def set_rotation_style(self, flag: _RotationStyle):
 304        self.rotation_style = flag
 305
 306    def flip_horizontal(self):
 307        self.flip_x = not self.flip_x
 308
 309    def flip_vertical(self):
 310        self.flip_y = not self.flip_y
 311
 312    def set_animation(self, new_mode):
 313        if new_mode == self.animation_name:
 314            return 
 315        self.animation_name = new_mode
 316        self.frames = self.frame_dict[new_mode]
 317        self.frame_idx = 0
 318
 319    def set_frame(self, idx):
 320        # also allow direct setting of frame_idx
 321        self.frame_idx = idx
 322
 323    def next_frame(self):
 324        self.frame_idx = (self.frame_idx+1) % len(self.frames)
 325
 326    # core transform requests 
 327    def set_scale(self, factor):
 328        self.scale_factor = factor
 329        self.request_transform = True
 330
 331    def set_brightness(self, factor):
 332        self.brightness_factor = factor
 333        self.request_transform = True
 334
 335    def set_transparency(self, factor):
 336        self.transparency_factor = factor
 337        self.request_transform = True
 338
 339    def blit_persist(self, surface: pygame.Surface, offset=(0,0), centre=True, reset=True):
 340        w, h = surface.get_width(), surface.get_height()
 341        if centre:
 342            offset = (offset[0]-w/2, offset[1]-h/2)
 343        if reset: 
 344            self.blit_surfaces[(self.animation_name, self.frame_idx)] = [(surface, offset)]
 345        else: 
 346            self.blit_surfaces[(self.animation_name, self.frame_idx)].append((surface, offset))
 347        self.request_transform = True
 348        
 349    # transform related helper
 350    def scale_by(self, factor):
 351        self.set_scale(self.scale_factor*factor)
 352
 353    def write_text(self, text: str, font: pygame.font.Font, colour=(255,255,255), offset=(0,0), centre=True, reset=True):
 354        text_surface = font.render(text, True, colour) 
 355        self.blit_persist(text_surface, offset, centre, reset)
 356
 357    # transform
 358    def transform_frames(self):
 359        self.request_transform = False
 360        for k, frames in self.frame_dict_original.items():
 361            new_frames = []
 362            for idx, f in enumerate(frames):
 363                f_new = f.copy()
 364                for s, o in self.blit_surfaces[(k, idx)]:
 365                    f_new.blit(s, o)
 366                f_new = set_transparency(f_new, self.transparency_factor)
 367                f_new = adjust_brightness(f_new, self.brightness_factor)
 368                f_new = pygame.transform.scale_by(f_new, self.scale_factor)
 369                new_frames.append(f_new)
 370
 371            self.frame_dict[k] = new_frames
 372            
 373        self.frames = self.frame_dict[self.animation_name]
 374
 375    def on_update(self, x, y, angle) -> Tuple[pygame.Surface, pygame.Rect, pygame.Mask]:
 376        if self.request_transform:
 377            self.transform_frames()
 378
 379        img = self.frames[self.frame_idx]
 380
 381        if self.rotation_style == _RotationStyle.ALL_AROUND: 
 382            img = pygame.transform.rotate(img, -angle)
 383    
 384        elif self.rotation_style == _RotationStyle.LEFTRIGHT:
 385            if angle > -90 and angle < 90:
 386                pass
 387            else:
 388                img = pygame.transform.flip(img, True, False)
 389             
 390        elif self.rotation_style == _RotationStyle.FIXED:
 391            pass
 392
 393        img = pygame.transform.flip(img, self.flip_x, self.flip_y)
 394
 395        self.image = img
 396
 397        img_w, img_h = img.get_width(), img.get_height()
 398        rect = img.get_rect(
 399            center=(x, y),
 400              width=img_w,
 401              height=img_h,
 402              )
 403        mask = pygame.mask.from_surface(self.image, 127)
 404        return img, rect, mask

 505class Sprite(pygame.sprite.Sprite):
 506    """
 507    Objects of the Sprite class represents a sprite.
 508    """
 509    def __init__(
 510            self, 
 511            frame_dict: Dict[str, List[pygame.Surface]], 
 512            starting_mode:Optional[str]=None, 
 513            position= (100, 100), 
 514            shape_type = ShapeType.BOX, 
 515            shape_size_factor=1.0, 
 516            body_type=pymunk.Body.KINEMATIC
 517        ):
 518        """
 519        You might not need to create the sprite from this constructor function. 
 520        Consider functions like `create_single_costume_sprite` or `create_animated_sprite`
 521        as they would be easier to work with. 
 522
 523        Example:
 524        ```python
 525        image1 = helper.load_image("assets/image1.png")
 526        image2 = helper.load_image("assets/image2.png")
 527        image3 = helper.load_image("assets/image3.png")
 528        image4 = helper.load_image("assets/image4.png")
 529
 530        frame_dict = {"walking": [image1, image2], "idling": [image3, image4]}
 531        my_sprite = Sprite(frame_dict, "walking", shape_type="circle", body_type=pymunk.Body.DYNAMIC)
 532        
 533        # alternative (exactly the same)
 534        my_sprite = Sprite(frame_dict, "walking", shape_type=ShapeType.CIRCLE, body_type=pymunk.Body.DYNAMIC)
 535        ```
 536
 537        Parameters
 538        ---
 539        frame_dict: Dict[str, List[pygame.Surface]]
 540            A dictionary with different frame modes (str) as the keys 
 541            and lists of images as the values 
 542
 543        starting_mode:Optional[str]
 544            The starting frame mode. If not provided, 
 545            any one of the frame mode might be picked 
 546            as the starting frame mode.
 547
 548        position: Tuple[float, float]
 549
 550        shape_type: ShapeType 
 551            The collision shape. See `set_shape` for more details.
 552        shape_size_factor: float 
 553
 554        body_type: int 
 555            The pymunk body type. Leave out the parameter if unsure. 
 556            Can be `pymunk.Body.KINEMATIC`, `pymunk.Body.DYNAMIC` or `pymunk.Body.STATIC` 
 557            - Use kinematic if you want the sprite to move when when you tell it to. 
 558            - Use dynamic if you want the sprite to be freely moving by physics. Also refer to `set_collision_type` to enable collision.  
 559            - Use static if you do not want it to move at all. 
 560        """
 561        super().__init__()
 562
 563        self.image: pygame.Surface # rotated and flipped every update during self.update
 564        "@private"
 565        self.rect: pygame.Rect # depends on the rotated image and thus should be redefined during self.update
 566        "@private"
 567
 568        if starting_mode is None:
 569            starting_mode = next(iter(frame_dict))
 570
 571        self._drawing_manager = _DrawingManager(frame_dict, starting_mode)
 572        _initial_frame = frame_dict[starting_mode][0]
 573        self._physics_manager = _PhysicsManager(game, body_type, shape_type, shape_size_factor, position,_initial_frame)
 574        
 575        self.private_data = {}
 576        """
 577        A dictionary similar to `game.shared_data`. 
 578        You can put any data or variable that should belong to the individuals sprite. 
 579        A good example would be the health point of a charactor. 
 580
 581        Let say if you have a uncertain number of enemy in game created by cloning or otherwise, 
 582        it would be messy to put the health point of each enemy to `game.shared_data`. In this 
 583        case, putting the health point in the private data is a better choice.
 584
 585        Example:
 586        ```python
 587        my_sprite.private_data['hp'] = 10

Let say if you have a uncertain number of enemy in game created by cloning or otherwise, it would be messy to put the health point of each enemy to game.shared_data. In this case, putting the health point in the private data is a better choice.

Example:

my_sprite.private_data['hp'] = 10

def on_hit(damage):
    my_sprite.private_data['hp'] -= damage
    print("how much hp I have left: ", my_sprite.private_data['hp'])

my_sprite.when_received_message('hit').add_handler(on_hit)
